from ibm_watsonx_orchestrate.agent_builder.tools import tool
import httpx

@tool
async def open_meteo_api(city: str, latitude: str = None, longitude: str = None) -> dict:
    """Fetches current weather for a given city using the Open-Meteo API.

    Args:
        city: The city name
        latitude: The city latitude (optional)
        longitude: The city longitude (optional)

    Returns:
        A dictionary containing the current weather data
    """
    if latitude and longitude:
        url = f"https://api.open-meteo.com/v1/forecast?latitude={latitude}&longitude={longitude}"
    else:
        url = f"https://api.open-meteo.com/v1/forecast?name={city}"
    async with httpx.AsyncClient() as client:
        response = await client.get(url)
        data = response.json()
        return {"result": data}
