from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json
import os

GITHUB_TOKEN = os.environ.get("GITHUB_TOKEN", "")

@tool
async def geocode(city: str) -> dict:
    """Geocode a city using Open-Meteo geocoding API.
    
    Args:
        city: City name
    
    Returns:
        Geocoding API response data
    """
    url = "https://api.open-meteo.com/v1/search?name=" + urllib.parse.quote(city) + "&count=1"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    return {"location": data["location"][0]}

# filename: fetch_weather.py
from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json
import os

@tool
async def fetch_weather(city: str) -> dict:
    """Fetch the current weather for a given city using the Open-Meteo API.
    
    Args:
        city: City name
    
    Returns:
        Current weather data
    """
    geocode_data = await geocode(city)
    lat = geocode_data["location"]["latitude"]
    lon = geocode_data["location"]["longitude"]
    url = f"https://api.open-meteo.com/v1/forecast?current_weather=true&latitude={lat}&longitude={lon}"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    return {"weather": data["current_weather"]}
