from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json

@tool
async def geocode_city(city: str) -> dict:
    """Geocode a city name to latitude and longitude using Open-Meteo.

    Args:
        city: The name of the city to geocode.

    Returns:
        A dictionary containing latitude, longitude, name, and country,
        or an error message if the city is not found.
    """
    params = urllib.parse.urlencode({"name": city, "count": 1})
    url = f"https://geocoding-api.open-meteo.com/v1/search?{params}"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    results = data.get("results", [])
    if not results:
        return {"error": f"No results found for city: {city}"}
    location = results[0]
    return {
        "latitude": location.get("latitude"),
        "longitude": location.get("longitude"),
        "name": location.get("name"),
        "country": location.get("country"),
    }
