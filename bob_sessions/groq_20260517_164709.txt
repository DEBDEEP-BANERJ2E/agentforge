```python
# filename: search_by_title.py
from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json

@tool
async def search_by_title(query: str, limit: int = 10) -> dict:
    """Search for books by title.

    Args:
        query: The search query
        limit: Maximum number of results (default: 10)

    Returns:
        A dictionary containing the search results
    """
    params = urllib.parse.urlencode({"q": query, "limit": limit})
    url = f"http://openlibrary.org/search.json?{params}"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    return {"items": data.get("docs", [])}
```

```python
from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json

@tool
async def fetch_by_id(book_id: str) -> dict:
    """Fetch full details by Open Library book ID.

    Args:
        book_id: The Open Library book ID

    Returns:
        A dictionary containing the book details
    """
    url = f"http://openlibrary.org/books/{book_id}.json"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    return {"book": data}
```
