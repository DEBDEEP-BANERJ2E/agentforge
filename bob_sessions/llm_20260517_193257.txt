from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json
import os

GITHUB_TOKEN = os.environ.get("GITHUB_TOKEN", "")

@tool
async def geocode_weather(city: str) -> dict:
    """Geocodes a city and fetches the current weather conditions.

    Args:
        city: The city to geocode

    Returns:
        A dictionary containing the geocoded city and current weather conditions
    """
    url = "https://geocoding-api.open-meteo.com/v1/search"
    params = urllib.parse.urlencode({"name": city, "count": 1})
    url = f"{url}?{params}"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    lat = data["results"][0]["latitude"]
    lon = data["results"][0]["longitude"]
    return {"lat": lat, "lon": lon}
