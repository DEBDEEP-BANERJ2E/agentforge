@tool
async def geocode_tool(city: str) -> dict:
    """Geocode city name to latitude/longitude using Open-Meteo API.
    
    Args:
        city: City name
    
    Returns:
        Geocoding API response data
    """
    url = "https://api.open-meteo.com/v1/search?name={}&count=1".format(urllib.parse.quote(city))
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    return {"lat": data["results"][0]["latitude"], "lon": data["results"][0]["longitude"]}

from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json

@tool
async def fetch_weather_tool(city: str) -> dict:
    """Fetch current weather for a given city using Open-Meteo API.
    
    Args:
        city: City name
    
    Returns:
        Weather API response data
    """
    geocode_data = await geocode_tool(city)
    lat = geocode_data["lat"]
    lon = geocode_data["lon"]
    url = f"https://api.open-meteo.com/v1/forecast?current_weather=true&latitude={lat}&longitude={lon}"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    return {"weather": data["current_weather"]}
