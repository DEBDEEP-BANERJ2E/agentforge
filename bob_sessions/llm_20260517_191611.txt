from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json
import os

GITHUB_TOKEN = os.environ.get("GITHUB_TOKEN", "")

@tool
async def geocode(city: str) -> dict:
    """Geocode a city to get its latitude and longitude.
    
    Args:
        city: The city name
    
    Returns:
        Geocoding API response data
    """
    url = "https://geocoding-api.open-meteo.com/v1/search"
    params = urllib.parse.urlencode({"name": city, "count": 1})
    url = f"{url}?{params}"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    return {"lat": data["results"][0]["latitude"], "lon": data["results"][0]["longitude"]}

from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json
import os

@tool
async def fetch_weather(city: str) -> dict:
    """Fetch the current weather for a given city.
    
    Args:
        city: The city name
    
    Returns:
        Weather API response data
    """
    geocode_result = await geocode(city)
    lat = geocode_result["lat"]
    lon = geocode_result["lon"]
    url = "https://api.open-meteo.com/v1/forecast"
    params = urllib.parse.urlencode({"latitude": lat, "longitude": lon})
    url = f"{url}?{params}"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    return {"weather": data}
