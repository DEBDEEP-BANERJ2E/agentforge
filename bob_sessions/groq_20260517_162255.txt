from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json

@tool
async def github_trending() -> dict:
    """Fetches the latest 5 GitHub trending repositories.

    Returns:
        A dictionary containing the names, stars, and descriptions of the trending repositories
    """
    url = "https://api.github.com/search/repositories?q=stars&sort=stars&order=desc&per_page=5"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    repos = []
    for item in data.get("items", []):
        repo = {
            "name": item.get("name", ""),
            "stars": item.get("stargazers_count", 0),
            "description": item.get("description", "")
        }
        repos.append(repo)
    return {"repositories": repos}
