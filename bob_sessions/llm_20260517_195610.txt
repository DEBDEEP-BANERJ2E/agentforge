from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json

@tool
async def fetch_weather(city: str) -> dict:
    """Fetch current weather conditions for a given city using the Open-Meteo API.

    Args:
        city: The city name

    Returns:
        A dictionary containing the current weather conditions
    """
    geocoding_url = f"https://geocoding-api.open-meteo.com/v1/search?name={urllib.parse.quote(city)}&count=1"
    with urllib.request.urlopen(geocoding_url, timeout=10) as resp:
        geocoding_data = json.loads(resp.read())
    results = geocoding_data.get("results", [])
    if not results:
        return {"error": f"Could not find location for city: {city}"}
    lat = results[0].get("latitude")
    lon = results[0].get("longitude")
    weather_url = f"https://api.open-meteo.com/v1/forecast?latitude={lat}&longitude={lon}&current_weather=true"
    with urllib.request.urlopen(weather_url, timeout=10) as resp:
        data = json.loads(resp.read())
    return {"weather": data.get("current_weather", {})}

from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json

@tool
async def fetch_air_quality(city: str) -> dict:
    """Fetch current air quality index for a given city using the Open-Meteo air quality API.

    Args:
        city: The city name

    Returns:
        A dictionary containing US AQI and PM2.5 values
    """
    geocoding_url = f"https://geocoding-api.open-meteo.com/v1/search?name={urllib.parse.quote(city)}&count=1"
    with urllib.request.urlopen(geocoding_url, timeout=10) as resp:
        geocoding_data = json.loads(resp.read())
    results = geocoding_data.get("results", [])
    if not results:
        return {"error": f"Could not find location for city: {city}"}
    lat = results[0].get("latitude")
    lon = results[0].get("longitude")
    aq_url = f"https://air-quality-api.open-meteo.com/v1/air-quality?latitude={lat}&longitude={lon}&hourly=pm2_5,us_aqi&timezone=auto&forecast_days=1"
    with urllib.request.urlopen(aq_url, timeout=10) as resp:
        data = json.loads(resp.read())
    hourly = data.get("hourly", {})
    times = hourly.get("time", [])
    aqi_values = hourly.get("us_aqi", [])
    pm25_values = hourly.get("pm2_5", [])
    return {
        "air_quality": {
            "us_aqi": aqi_values[0] if aqi_values else None,
            "pm2_5": pm25_values[0] if pm25_values else None,
            "time": times[0] if times else None,
        }
    }
