from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json
import os

GITHUB_TOKEN = os.environ.get("GITHUB_TOKEN", "")

@tool
async def geocode(city: str) -> dict:
    """Geocode a city to get its latitude and longitude.
    
    Args:
        city: The city name
    
    Returns:
        Geocoding API response data
    """
    url = f"https://geocoding-api.open-meteo.com/v1/search?name={urllib.parse.quote(city)}&count=1"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    return {"location": data["results"][0]["latitude"], "longitude": data["results"][0]["longitude"]}

# filename: fetch_weather.py
from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json
import os

@tool
async def fetch_weather(city: str) -> dict:
    """Fetch the current weather for a given city.
    
    Args:
        city: The city name
    
    Returns:
        Current weather data
    """
    location = await geocode(city)
    url = f"https://api.open-meteo.com/v1/forecast?current_weather=true&latitude={location['location']}&longitude={location['longitude']}"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    return {"weather": data["current_weather"]}
