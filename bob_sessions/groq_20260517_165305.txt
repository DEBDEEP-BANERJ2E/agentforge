```python
from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json

@tool
async def github_commits(repo_owner: str, repo_name: str, limit: int = 10) -> dict:
    """Fetch the latest commits from a GitHub repository.

    Args:
        repo_owner: The owner of the repository
        repo_name: The name of the repository
        limit: Maximum number of commits to fetch (default: 10)

    Returns:
        A dictionary with the latest commits
    """
    params = urllib.parse.urlencode({"per_page": limit})
    url = f"https://api.github.com/repos/{repo_owner}/{repo_name}/commits?{params}"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    return {"commits": data}
```

```python
from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json

@tool
async def github_ci_checks(repo_owner: str, repo_name: str, limit: int = 10) -> dict:
    """Fetch the failed CI checks from a GitHub repository.

    Args:
        repo_owner: The owner of the repository
        repo_name: The name of the repository
        limit: Maximum number of CI checks to fetch (default: 10)

    Returns:
        A dictionary with the failed CI checks
    """
    params = urllib.parse.urlencode({"per_page": limit})
    url = f"https://api.github.com/repos/{repo_owner}/{repo_name}/checks?{params}"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    failed_checks = [check for check in data if check.get("status", "") == "failure"]
    return {"failed_checks": failed_checks}
```
