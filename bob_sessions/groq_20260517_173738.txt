from ibm_watsonx_orchestrate.agent_builder.tools import tool
import urllib.request
import urllib.parse
import json

@tool
async def rest_countries_api(country_name: str) -> dict:
    """Looks up a country by name and returns its capital, population, region, and currencies.

    Args:
        country_name: The name of the country to look up

    Returns:
        A dictionary with the country's information
    """
    url = f"https://restcountries.com/v3.1/name/{urllib.parse.quote(country_name)}"
    with urllib.request.urlopen(url, timeout=10) as resp:
        data = json.loads(resp.read())
    if data:
        country = data[0]
        return {
            "capital": country.get("capital", [])[0] if country.get("capital") else "",
            "population": country.get("population", 0),
            "region": country.get("region", ""),
            "currencies": {k: v.get("name", "") for k, v in country.get("currencies", {}).items()}
        }
    else:
        return {"error": "Country not found"}
